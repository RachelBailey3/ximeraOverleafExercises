\begin{exercise}
Let $p(v) = 7 \sqrt[4]{-v}$. Compute:

\[
\frac{d}{dv}p(v)
\begin{prompt}=\answer{-\frac{7}{4 (-v)^{3/4}}}\end{prompt}
\]
\end{exercise}

\begin{exercise}
Let $y(u) = \cot \left(6 \sqrt{-u}\right)$. Compute:

\[
y'(u)
\begin{prompt}=\answer{\frac{3 \csc ^2\left(6 \sqrt{-u}\right)}{\sqrt{-u}}}\end{prompt}
\]
\end{exercise}
